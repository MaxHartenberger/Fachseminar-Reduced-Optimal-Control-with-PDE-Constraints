\section{Problem Formulation}

\begin{frame}{Problem formulation}
Goal: Find a control $u$ such that the state $y$ matches a desired profile $y_d$.

\medskip
PDE constraint:
\[
  -\Delta y = \chi_{\omega}\,u \quad \text{in }\Omega,\qquad y=0\quad\text{on }\partial\Omega.
\]

Cost:
\[
  J(y,u)=\tfrac12\|y-y_d\|_{L^2(\Omega)}^2+\tfrac{\beta}{2}\|u\|_{L^2(\Omega)}^2,\qquad \beta>0.
\]

\begin{itemize}
  \item $\Omega=[0,1]^2$, control acts only on a subdomain $\omega\subset\Omega$.
  \item Spaces: $V=H_0^1(\Omega)$, $U=L^2(\Omega)$.
\end{itemize}

In our computations, $\omega$ is the disk
\[
  \omega := \big\{x\in\Omega : (x_1-0.5)^2+(x_2-0.5)^2 \le 0.1^2\big\}.
\]
\end{frame}

\begin{frame}{Target potential $y_d$}
\[
  y_d(x)=0.5-\max\big(|x_1-0.5|,\,|x_2-0.5|\big).
\]

\includegraphics[height=0.62\textheight,keepaspectratio]{\meshplotsdir/y_d_3d.png}
\end{center}
\end{frame}

\section{Variational Formulation and Discretization}

\begin{frame}{Variational formulation}
Define
\begin{align*}
  a(y,v) &= \int_{\Omega} \nabla y\cdot\nabla v\, dx\quad\text{for }y,v\in V,\\
  b(u,v) &= \int_{\omega} u\, v\, dx\quad\text{for }u\in U,\ v\in V.
\end{align*}
State: given $u\in U$, find $y\in V$ such that
\[
  a(y,v)=b(u,v)\quad\forall v\in V.
\]
Objective:
\[
  J(y,u)=\tfrac12\int_\Omega (y-y_d)^2\,dx+\tfrac{\beta}{2}\int_\Omega u^2\,dx.
\]
\end{frame}

\begin{frame}{Discretization and discrete objective}
Let $V_h\subset H_0^1(\Omega)$ be the conforming P1 space with basis $(\varphi_i)_{i=1}^n$.\\

\centering
\includegraphics[height=0.74\textheight,keepaspectratio]{\meshdir/mesh_h_\meshhtoken/mesh.png}
\\
{\footnotesize Triangulation with subdomain $\omega$.}
\end{frame}

\begin{frame}{Discretization and discrete objective}
Assemble
\begin{align*}
  A_{ij}&=\int_\Omega \nabla\varphi_j\cdot\nabla\varphi_i\,dx, &
  M_{ij}&=\int_\Omega \varphi_j\varphi_i\,dx,\\
  B_{ij}&=\int_\omega \varphi_j\varphi_i\,dx, &
  (M_U)_{ij}&=\int_\Omega \varphi_j\varphi_i\,dx.
\end{align*}

Discrete state: 
\[
  A\bm y=B\bm u.
\]

Discrete objective:
\[
  J(\bm y,\bm u)=\tfrac12(\bm y-\bm y_d)^\top M(\bm y-\bm y_d)+\tfrac{\beta}{2}\bm u^\top M_U\bm u.
\]
\end{frame}

\section{Reduced Problem (Eliminate State)}

\begin{frame}{Reduced functional $F(\bm u)$}
Eliminate the state via the discrete PDE constraint
\[
  A\bm y = B\bm u,\qquad \bm y(\bm u)=A^{-1}B\bm u.
\]
Insert $\bm y(\bm u)$ into $J$ to obtain the reduced objective
\[
  F(\bm u)=\tfrac12\big(\bm y(\bm u)-\bm y_d\big)^\top M\big(\bm y(\bm u)-\bm y_d\big)
  +\tfrac{\beta}{2}\,\bm u^\top M_U\bm u.
\]
\end{frame}

\begin{frame}{$U$-gradient and Riesz map}
We equip $U_h\cong\mathbb{R}^n$ with the mass-matrix inner product
\[
  \langle \bm a,\bm b\rangle_U:=\bm a^\top M_U\bm b,\qquad \|\bm a\|_U:=\sqrt{\bm a^\top M_U\bm a}.
\]
The $U$-gradient $\nabla_U F(\bm u)$ is defined by
\[
  \langle \nabla_U F(\bm u),\bm v\rangle_U = F'(\bm u)[\bm v]\quad\forall\bm v.
\]
If $\nabla F(\bm u)$ denotes the coordinate gradient (Euclidean inner product), then
\[
  \nabla F(\bm u)=M_U\,\nabla_U F(\bm u).
\]
\end{frame}

\begin{frame}{Adjoint formula}
Eliminate state: $\bm y(\bm u)=A^{-1}B\bm u$. Solve
\[
  A\bm y=B\bm u,\qquad A\bm p=M(\bm y-\bm y_d).
\]
Then
\[
  \nabla F(\bm u)=B^\top\bm p + \beta M_U\bm u,\qquad
  \nabla_U F(\bm u)=M_U^{-1}B^\top\bm p + \beta\bm u.
\]
\end{frame}

\begin{frame}{Optimality condition}
The reduced functional $F(\bm u)$ is strictly convex, hence has a unique minimizer $\bm u_*$. The first-order condition is
\[
  \nabla_U F(\bm u_*)=\bm 0.
\]
Equivalently (multiply by $M_U$),
\[
  \nabla F(\bm u_*)=\bm 0.
\]
\end{frame}

\begin{frame}{Convex quadratic problem formulation}
Expand $F$ as an unconstrained quadratic problem
\[
  F(\bm u)=\tfrac12\bm u^\top Q\bm u - \bm b^\top\bm u + \text{const},
\]
with
\[
  Q=B^\top A^{-1}MA^{-1}B + \beta M_U\quad\text{(SPD)},\qquad \bm b=B^\top A^{-1}M\bm y_d.
\]
Stationarity system:
\[
  Q\bm u_* = \bm b.
\]
\end{frame}

\begin{frame}{Lipschitz constant $L$ and strong convexity $m$}
In the $U$-inner product, the reduced Hessian is
\[
  H = M_U^{-1}Q \quad\text{(self-adjoint and SPD in }\langle\cdot,\cdot\rangle_U\text{)}.
\]
Its extremal eigenvalues satisfy the generalized eigenproblem
\[
  Q\bm v = \lambda\,M_U\bm v.
\]
Hence
\[
  L=\lambda_{\max},\qquad m=\lambda_{\min},\qquad \kappa=\frac{L}{m}.
\]
These constants define the fixed GD step $\alpha=1/L$ and Nesterov parameters.
\end{frame}

\section{Optimization Methods}

\begin{frame}{Gradient descent (GD)}
Iteration (fixed step):
\[
  \bm u_{k+1}=\bm u_k-\tfrac{1}{L}\,\nabla_U F(\bm u_k).
\]

\medskip
Stopping criterion:
\[
  \|\nabla_U F(\bm u_k)\|_U \le \texttt{tol}_{abs}+\texttt{tol}_{rel}\,\|\nabla_U F(\bm 0)\|_U,
\]
with $\texttt{tol}_{abs}=10^{-5}$ and $\texttt{tol}_{rel}=10^{-3}$.
\end{frame}

\begin{frame}{Barzilai--Borwein (BB)}
Idea: Use a secant-based scalar step size in the $U$-metric.

\medskip
With $\bm s_k=\bm u_k-\bm u_{k-1}$ and $\bm d_k=\nabla_U F(\bm u_k)-\nabla_U F(\bm u_{k-1})$,
\[
  \alpha_k^{\mathrm{BB1}}=\frac{\langle \bm d_k,\bm d_k\rangle_U}{\langle \bm s_k,\bm d_k\rangle_U},
  \qquad
  \alpha_k^{\mathrm{BB2}}=\frac{\langle \bm s_k,\bm s_k\rangle_U}{\langle \bm s_k,\bm d_k\rangle_U}.
\]

\medskip
Update: $\bm u_{k+1}=\bm u_k-\alpha_k\nabla_U F(\bm u_k)$ (alternating BB1/BB2).

\medskip
Stopping criterion: same $\|\nabla_U F\|_U$ rule as GD.
\end{frame}

\begin{frame}{Nesterov acceleration}
With $\kappa=L/m$, define
\[
  \beta_N=\frac{\sqrt{\kappa}-1}{\sqrt{\kappa}+1},\qquad
  \bm y_k=\bm u_k+\beta_N(\bm u_k-\bm u_{k-1}).
\]
Update (fixed step):
\[
  \bm u_{k+1}=\bm y_k-\tfrac{1}{L}\,\nabla_U F(\bm y_k).
\]
Stopping criterion: same $\|\nabla_U F\|_U$ rule as GD.
\end{frame}

\begin{frame}{Conjugate gradients (CG) for the stationarity system}
Solve the SPD linear system
\[
  Q\bm u=\bm b,\qquad \bm b=B^\top A^{-1}M\bm y_d.
\]

Stopping criterion:
\[
  \|\bm r_k\|_2\le \texttt{rtol}\,\|\bm b\|_2,\qquad \texttt{rtol}=10^{-4},\qquad \bm r_k=\bm b-Q\bm u_k.
\]

\medskip
Link to the $U$-gradient: since $\nabla F(\bm u_k)=\bm r_k$ for this quadratic problem,
\[
  \|\nabla_U F(\bm u_k)\|_U = \sqrt{\bm r_k^\top M_U^{-1}\bm r_k}.
\]
\end{frame}

\begin{frame}{Interpretation}
\begin{itemize}
  \item The reduced problem is a strictly convex quadratic program, so all methods converge to the same minimizer.
  \item The gradient-norm histories show rapid decay for BB, GD ($1/L$), and Nesterov; on this mesh BB and GD nearly overlap.
  \item Runtime reflects a trade-off between iterations and preprocessing: GD needs an $L$ estimate and Nesterov needs both $(m,L)$ (costs $t_L$ and $t_m$).
  \item BB and CG avoid spectral estimates; on the representative mesh, BB achieves the smallest wall-clock time.
\end{itemize}
\end{frame}

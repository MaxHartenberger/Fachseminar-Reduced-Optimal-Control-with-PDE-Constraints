\section{Problem Statement}
We consider a linear-quadratic optimal control problem governed by the Poisson equation on a polygonal domain \(\Omega\subset\mathbb{R}^2\) (in computations, typically \([0,1]^2\)). A circular subregion \(\omega\subset\Omega\) is tagged on the mesh and serves as the support of the control. The control acts as a volumetric source term restricted to \(\omega\):
\begin{align}
  -\Delta y &= \chi_{\omega}\, u \quad\text{in } \Omega,\\
  y &= 0 \quad\text{on } \partial\Omega,
\end{align}
where \(\chi_{\omega}\) denotes the characteristic function of \(\omega\). We denote by \(V=H_0^1(\Omega)\) the state space and by \(U=L^2(\Omega)\) the control space. For a given target \(y_d\in L^2(\Omega)\) and a regularization parameter \(\beta>0\), we minimize
\begin{equation}
  J(y,u) = \tfrac12 \int_{\Omega} (y - y_d)^2\,dx + \tfrac{\beta}{2} \int_{\Omega} u^2\,dx.
\end{equation}
The target is chosen as the pyramid-like profile
\begin{equation}
  y_d(x) = 0.5 - \max\big(|x_1-0.5|,\,|x_2-0.5|\big),
\end{equation}
which belongs to \(L^2(\Omega)\). The control enters only on \(\omega\) via \(\chi_{\omega}\), so the effective source equals \(u\) on \(\omega\) and vanishes on \(\Omega\setminus\omega\). The PDE with homogeneous Dirichlet boundary conditions is well-posed for \(u\in L^2(\Omega)\), yielding a unique weak solution \(y\in V\) by Lax--Milgram.

\section{Variational Formulation and Discretization}
\paragraph{Weak formulation.} Define the bilinear and linear forms
\begin{align}
  a(y,v) &= \int_{\Omega} \nabla y\cdot\nabla v\, dx, & b(u,v) &= \int_{\omega} u\, v\, dx.
\end{align}
For any \(u\in L^2(\Omega)\), the weak state equation reads: find \(y\in V\) such that
\begin{equation}
  a(y,v) = b(u,v) \quad \forall v\in V.
\end{equation}
By coercivity of \(a(\cdot,\cdot)\) on \(V\) and boundedness of \(b(\cdot,\cdot)\), this admits a unique \(y\in V\).

\paragraph{Finite elements.} 
Let \(\mathcal{T}_h\) be a conforming triangulation of \(\Omega\). We use the conforming P1 space \(V_h\subset V\) with nodal basis \(\{\varphi_i\}_{i=1}^{n}\). The control is represented in the same P1 basis but measured with the \(L^2\) inner product (``control metric''). We assemble the matrices
\begin{align}
  A_{ij} &= \int_{\Omega} \nabla \varphi_j\cdot\nabla \varphi_i\, dx, &
  M_{ij} &= \int_{\Omega} \varphi_j\, \varphi_i\, dx, &
  B_{ij} &= \int_{\omega} \varphi_j\, \varphi_i\, dx, &
  (M_U)_{ij} &= \int_{\Omega} \varphi_j\, \varphi_i\, dx.
\end{align}
Here, \(A\) is the stiffness matrix for state/adjoint solves, \(M\) is the state mass matrix used to weight the tracking term, $B$ maps control coefficients to the state RHS via restriction to \(\omega\), and $M_U$ is the control-space mass matrix defining the \(L^2\) inner product in \(U_h\).

\paragraph{Boundary conditions.} 
Homogeneous Dirichlet boundary conditions are imposed on the state/adjoint by modifying \(A\) and zeroing the RHS at boundary degrees of freedom when solving. Importantly, no boundary conditions are applied to \(B\) or $M_U$; these operators remain the unmodified \(L^2\)-based forms on \(\Omega\).

\paragraph{Discrete state equation and objective.} 
With coefficient vectors \(\mathbf{y},\mathbf{u}\in\mathbb{R}^n\) and target coefficients \(\mathbf{y}_d\in\mathbb{R}^n\), the discrete state equation and cost read
\begin{align}
  A\, \mathbf{y} &= B\, \mathbf{u}, \\
  J(\mathbf{y},\mathbf{u}) &= \tfrac12 (\mathbf{y}-\mathbf{y}_d)^\top M (\mathbf{y}-\mathbf{y}_d) + \tfrac{\beta}{2} \mathbf{u}^\top M_U\, \mathbf{u}.
\end{align}

\section{Reduced Formulation}
\paragraph{State elimination.} Eliminating the state via the discrete PDE constraint gives \(\mathbf{y}(\mathbf{u})=A^{-1}B\,\mathbf{u}\) and the reduced functional \(F\colon\mathbb{R}^n\to\mathbb{R}\):
\begin{equation}
  F(\mathbf{u}) = \tfrac12 \big(A^{-1} B\, \mathbf{u} - \mathbf{y}_d\big)^\top M \big(A^{-1} B\, \mathbf{u} - \mathbf{y}_d\big) + \tfrac{\beta}{2}\, \mathbf{u}^\top M_U\, \mathbf{u}.
\end{equation}

\paragraph{Adjoint and gradient (continuous view).} Introducing an adjoint variable \(p\in V\) and the Lagrangian
\begin{equation}
  \mathcal{L}(y,u,p) = \tfrac12\|y-y_d\|_{L^2(\Omega)}^2 + \tfrac{\beta}{2}\|u\|_{L^2(\Omega)}^2 + a(y,p) - b(u,p),
\end{equation}
the optimality system reads: state \(a(y,v)=b(u,v)\), adjoint \(a(v,p)=(y-y_d,v)\) for all \(v\in V\), and variational gradient condition \((\beta u + \chi_{\omega} p, w)_{L^2(\Omega)}=0\) for all \(w\in U\). Hence, formally, \(\beta u + p\,\chi_{\omega}=0\) in \(\Omega\).

\paragraph{Adjoint and gradient (discrete view).} In coefficients, the adjoint solves
\begin{equation}
  A\, \mathbf{p} = M\, (\mathbf{y}(\mathbf{u}) - \mathbf{y}_d),
\end{equation}
and the gradient with respect to the control space inner product induced by \(M_U\) is
\begin{equation}
  \nabla F(\mathbf{u}) = M_U^{-1}\, B^\top \mathbf{p} + \beta\, \mathbf{u}.
\end{equation}
The first-order optimality condition \(\nabla F(\mathbf{u}_*)=0\) is equivalent to
\begin{equation}
  B^\top \mathbf{p}_* + \beta\, M_U\, \mathbf{u}_* = 0, \qquad A\, \mathbf{p}_* = M\, (\mathbf{y}(\mathbf{u}_*)-\mathbf{y}_d).
\end{equation}

\paragraph{Hessian, convexity, and Lipschitz constant.} The reduced Hessian applied to \(\mathbf{v}\) is
\begin{equation}
  H\, \mathbf{v} = M_U^{-1}\, B^\top A^{-1} M A^{-1} B\, \mathbf{v} + \beta\, \mathbf{v}.
\end{equation}
Since \(A\), \(M\), and \(M_U\) are symmetric positive definite on their natural subspaces and \(B\) is linear, the quadratic form associated with \(H\) is strictly positive for \(\beta>0\); thus, \(F\) is strictly convex with a unique minimizer. In the \(U\)-metric, the gradient is Lipschitz with constant
\begin{equation}
  L = \big\| M_U^{-1/2} B^\top A^{-1} M A^{-1} B\, M_U^{-1/2} \big\|_2 + \beta,
\end{equation}
namely the spectral radius of the SPD term plus \(\beta\).

\section{Optimization Methods}
We first rewrite the reduced problem as a quadratic optimization problem in the control coefficients. Using the discrete operators, the objective
\[
  F(\mathbf{u}) = \tfrac12 \big(A^{-1} B\, \mathbf{u} - \mathbf{y}_d\big)^\top M \big(A^{-1} B\, \mathbf{u} - \mathbf{y}_d\big) + \tfrac{\beta}{2}\, \mathbf{u}^\top M_U\, \mathbf{u}
\]
expands to the unconstrained QP
\begin{equation}
  \min_{\mathbf{u}\in\mathbb{R}^n} \; \tfrac12\, \mathbf{u}^\top Q\, \mathbf{u} + \mathbf{c}^\top \mathbf{u} + \text{const},
  \qquad Q = B^\top A^{-1} M A^{-1} B + \beta\, M_U,\quad \mathbf{c} = -\, B^\top A^{-1} M\, \mathbf{y}_d,
\end{equation}
with constant term $\text{const} = \tfrac12\, \mathbf{y}_d^\top M\, \mathbf{y}_d$. Since $A$, $M$, and $M_U$ are SPD on their natural subspaces, $Q$ is SPD, so the QP is strictly convex with unique minimizer characterized by the normal equations
\begin{equation}
  Q\, \mathbf{u}_* = B^\top A^{-1} M\, \mathbf{y}_d.
\end{equation}

\paragraph{Equivalence with reduced KKT.}
Rearranging the normal equations gives
\[
  \beta\, M_U\, \mathbf{u}_* \;+\; B^\top A^{-1} M\,\big(A^{-1} B\, \mathbf{u}_* - \mathbf{y}_d\big) \;=\; 0.
\]
Defining the adjoint by $A\, \mathbf{p}_* = M\,\big(A^{-1} B\, \mathbf{u}_* - \mathbf{y}_d\big)$ (i.e., $\mathbf{p}_* = A^{-1} M\,(A^{-1} B\, \mathbf{u}_* - \mathbf{y}_d)$) yields the reduced optimality condition
\[
  B^\top \mathbf{p}_* + \beta\, M_U\, \mathbf{u}_* = 0,
\]
which is exactly the KKT stationarity written in Section 3.

We solve the unconstrained, strictly convex reduced problem in the control space endowed with the \(L^2\) inner product induced by \(M_U\):
\begin{equation}
  \min_{\mathbf{u}\in\mathbb{R}^n} \; F(\mathbf{u}) = \tfrac12 (A^{-1} B\, \mathbf{u} - \mathbf{y}_d)^\top M (A^{-1} B\, \mathbf{u} - \mathbf{y}_d) + \tfrac{\beta}{2}\, \mathbf{u}^\top M_U\, \mathbf{u}.
\end{equation}
The control-space inner product and norm are
\begin{equation}
  \langle \mathbf{a}, \mathbf{b} \rangle_U := \mathbf{a}^\top M_U\, \mathbf{b}, \qquad \|\mathbf{a}\|_U := \sqrt{\langle \mathbf{a}, \mathbf{a} \rangle_U}.
\end{equation}
The gradient used by all methods is the Riesz-represented gradient in \(U\):
\begin{equation}
  \nabla F(\mathbf{u}) = M_U^{-1}\, B^\top \mathbf{p} + \beta\, \mathbf{u}, \quad \text{where } A\, \mathbf{p} = M\, (\mathbf{y}(\mathbf{u}) - \mathbf{y}_d),\; A\, \mathbf{y}(\mathbf{u}) = B\, \mathbf{u}.
\end{equation}
Unless stated otherwise, we initialize at \(\mathbf{u}_0 = \nabla F(\mathbf{0})\) (as in the code) and stop when \(\|\nabla F(\mathbf{u}_k)\|_U \le \texttt{tol}\) or a maximum number of iterations is reached.

\subsection{Gradient Descent with Fixed Step \(\alpha = 1/L\)}
For an \(L\)-Lipschitz gradient in the \(U\)-metric, the fixed-step update
\begin{equation}
  \mathbf{u}_{k+1} = \mathbf{u}_k - \alpha\, \nabla F(\mathbf{u}_k), \quad \alpha = 1/L,
\end{equation}
produces a monotone decrease in \(F\) for convex \(F\). In practice, \(L\) is estimated via a matrix-free power iteration on the Hessian action
\(H\,\mathbf{v} = M_U^{-1} B^\top A^{-1} M A^{-1} B\, \mathbf{v} + \beta\, \mathbf{v}\).

\paragraph{Spectral viewpoint (how $L$ and $m$ are computed).}
In the $U$-metric, $H$ is self-adjoint and SPD, and can be written as $H = M_U^{-1} Q$ with
\(Q = B^\top A^{-1} M A^{-1} B + \beta\, M_U\). Therefore, the extreme eigenvalues of $H$ in the $U$-inner product are the extreme \emph{generalized} eigenvalues of the pencil $(Q,M_U)$:
\begin{equation}
  Q\,\mathbf{v} = \lambda\, M_U\,\mathbf{v}.
\end{equation}
The Lipschitz constant is $L=\lambda_{\max}$ and the strong convexity constant is $m=\lambda_{\min}$. Numerically, we approximate $L$ either by power iteration applied to $H$ using the $U$-Rayleigh quotient, or by solving the generalized eigenproblem for the largest eigenpair; for $(m,L)$ we solve the generalized eigenproblem for both extremes.

\begin{algorithm}
\caption{Gradient Descent in $U$ with step $\alpha=1/L$}\label{alg:gd}
\KwData{tolerance $\texttt{tol}>0$, $\texttt{max\_iter}$, model with $\nabla F$, $\|\cdot\|_U$}
\KwResult{sequence $(\mathbf{u}_k)$ converging to a minimizer}
$\mathbf{u} \gets \mathbf{u}_0$\; \tcp*[r]{default $\nabla F(\mathbf{0})$}
Estimate $L$ by power iteration; $\alpha \gets 1/L$\;
\For{$k=0,1,\dots$}{
  $\mathbf{g} \gets \nabla F(\mathbf{u})$; $g_n \gets \|\mathbf{g}\|_U$\;
  \If{$g_n \le \texttt{tol}$}{break}
  $\mathbf{u} \gets \mathbf{u} - \alpha\, \mathbf{g}$\;
}
\end{algorithm}

\subsection{Barzilai--Borwein (Alternating BB1/BB2)}
BB methods approximate inverse curvature along the last step by selecting the steplength from secant information in the \(U\)-inner product. Define \(\mathbf{s}_k=\mathbf{u}_k-\mathbf{u}_{k-1}\), \(\mathbf{d}_k=\nabla F(\mathbf{u}_k)-\nabla F(\mathbf{u}_{k-1})\), and the \(U\)-inner product \(\langle \cdot, \cdot \rangle_U\). The two classical choices are
\begin{align}
  \alpha_k^{\text{BB1}} &= \frac{\langle \mathbf{d}_k, \mathbf{d}_k \rangle_U}{\langle \mathbf{s}_k, \mathbf{d}_k \rangle_U}, &
  \alpha_k^{\text{BB2}} &= \frac{\langle \mathbf{s}_k, \mathbf{s}_k \rangle_U}{\langle \mathbf{s}_k, \mathbf{d}_k \rangle_U}.
\end{align}
We alternate BB1/BB2 per iteration and fall back to \(\alpha_k=1\) if \(\big|\langle \mathbf{s}_k, \mathbf{d}_k \rangle_U\big|\) is numerically tiny. The update is
\(\mathbf{u}_{k+1} = \mathbf{u}_k - (1/\alpha_k)\, \nabla F(\mathbf{u}_k)\).

\begin{algorithm}
\caption{Barzilai--Borwein in $U$ (alternating BB1/BB2)}\label{alg:bb}
\KwData{$\texttt{tol}>0$, $\texttt{max\_iter}$, model with $\nabla F$, $\langle \cdot, \cdot \rangle_U$, $\|\cdot\|_U$}
\KwResult{sequence $(\mathbf{u}_k)$}
$\mathbf{u}_{-1} \gets \mathbf{0}$; $\mathbf{g}_{-1} \gets \nabla F(\mathbf{0})$\;
$\mathbf{u}_0 \gets \mathbf{u}_0$\; $\mathbf{g}_0 \gets \nabla F(\mathbf{u}_0)$\; \tcp*[r]{default $\mathbf{g}_{-1}$}
\For{$k=0,1,\dots$}{
  \If{$\|\mathbf{g}_k\|_U \le \texttt{tol}$}{break}
  $\mathbf{s}_k \gets \mathbf{u}_k - \mathbf{u}_{k-1}$; $\mathbf{d}_k \gets \mathbf{g}_k - \mathbf{g}_{k-1}$; $s\!d \gets \langle \mathbf{s}_k, \mathbf{d}_k \rangle_U$\;
  \eIf{$|s\!d| < \varepsilon$}{
    $\alpha_k \gets 1$\;
  }{
    \eIf{$k$ is even}{
      $\alpha_k \gets \langle \mathbf{d}_k, \mathbf{d}_k \rangle_U / s\!d$\;
    }{
      $\alpha_k \gets s\!d / \max\{\langle \mathbf{s}_k, \mathbf{s}_k \rangle_U, \varepsilon\}$\;
    }
  }
  $\mathbf{u}_{k+1} \gets \mathbf{u}_k - (1/\alpha_k)\, \mathbf{g}_k$\;
  $\mathbf{u}_{k-1} \gets \mathbf{u}_k$; $\mathbf{g}_{k-1} \gets \mathbf{g}_k$\;
  $\mathbf{u}_k \gets \mathbf{u}_{k+1}$; $\mathbf{g}_k \gets \nabla F(\mathbf{u}_k)$\;
}
\end{algorithm}


\subsection{Nesterov Acceleration with Known $(m,L)$ (Constant Parameters)}
For the strongly convex quadratic QP with Hessian $Q$ in the $U$-metric, let
$L=\lambda_{\max}$ and $m=\lambda_{\min}$ denote the largest/smallest generalized eigenvalues of the pencil $(Q, M_U)$,
equivalently the spectrum of $H = M_U^{-1/2} Q M_U^{-1/2}$. Set
\[\alpha = 1/L,\qquad \kappa=L/m,\qquad \beta = \frac{\sqrt{\kappa}-1}{\sqrt{\kappa}+1}.\]
Then the constant-parameter Nesterov scheme reads
\begin{align}
  \mathbf{y}_k &= \mathbf{u}_k + \beta\,(\mathbf{u}_k - \mathbf{u}_{k-1}), &
  \mathbf{u}_{k+1} &= \mathbf{y}_k - \alpha\, \nabla F(\mathbf{y}_k),
\end{align}
and achieves a linear rate $\|\mathbf{u}_k-\mathbf{u}_*\|_U \le C\,\big(\tfrac{\sqrt{\kappa}-1}{\sqrt{\kappa}+1}\big)^k$.
In practice, we compute $(L,m)$ by solving the generalized eigenproblem $(Q, M_U)$ with $Q\,v = \lambda\, M_U\,v$, where
$Q v = B^T A^{-1} M A^{-1} B v + \beta M_U v$ is applied matrix-free via $Q v = M_U\, H v$.

\begin{algorithm}
\caption{Nesterov (constant parameters)}\label{alg:nest-ml}
\KwData{$\texttt{tol}>0$, $\texttt{max\_iter}$, model with $\nabla F$, $M_U$, Hessian action $H$}
\KwResult{sequence $(\mathbf{u}_k)$}
Estimate $(L,m)$ from generalized eigenpairs of $(Q, M_U)$; $\alpha \gets 1/L$; $\kappa \gets L/m$; $\beta \gets (\sqrt{\kappa}-1)/(\sqrt{\kappa}+1)$

$\mathbf{u}_{-1} \gets \mathbf{u}_0$; $\mathbf{u} \gets \mathbf{u}_0$\;
\For{$k=0,1,\dots$}{
  $\mathbf{y} \gets \mathbf{u} + \beta (\mathbf{u} - \mathbf{u}_{k-1})$\; $\mathbf{g} \gets \nabla F(\mathbf{y})$; $g_n \gets \|\mathbf{g}\|_U$\;
  \If{$g_n \le \texttt{tol}$}{break}
  $\mathbf{u}_{k+1} \gets \mathbf{y} - \alpha\, \mathbf{g}$\;
}
\end{algorithm}

\subsection{Conjugate Gradient (CG) for the Stationarity System}
The reduced problem is strictly convex and quadratic. The first-order optimality condition $\nabla F(\mathbf{u}_*)=0$ can be written as an SPD linear system in the control unknowns,
\begin{equation}
  Q\,\mathbf{u}_* = B^\top A^{-1} M\,\mathbf{y}_d, \qquad Q := B^\top A^{-1} M A^{-1} B + \beta\, M_U.
\end{equation}
In the implementation, we solve this system with Conjugate Gradient in a matrix-free manner (Hessian action only). Concretely, we apply $Q\mathbf{v}$ via the reduced Hessian action and the control metric, and call SciPy's routine \texttt{scipy.sparse.linalg.cg} on a \texttt{LinearOperator}.

\begin{algorithm}[]
\caption{CG for $Q\mathbf{u}=\mathbf{b}$ (matrix-free)}\label{alg:cg}
\KwData{model with $A^{-1}$ solves, matrices $B,M,M_U$, Hessian action $H$; $\texttt{rtol}>0$}
\KwResult{$\mathbf{u}_*$}
$\mathbf{b} \gets B^\top A^{-1}(M\,\mathbf{y}_d)$\;
Define operator $Q\mathbf{v} := M_U\,(H\mathbf{v})$ (equivalently $B^\top A^{-1} M A^{-1} B\,\mathbf{v} + \beta\,M_U\mathbf{v}$)\;
Call CG to solve $Q\mathbf{u}=\mathbf{b}$ to relative tolerance $\texttt{rtol}$\;
\end{algorithm}

\subsection{Remarks on Metrics and Implementation}
All inner products and norms above are in the control metric induced by \(M_U\). This choice ensures that steplengths and convergence tests reflect the natural topology of the control space (consistent with regularization). The adjoint reuse of \(A\) facilitates efficient linear solves and preconditioning.

\section{Numerical Results}
We evaluate three first-order methods on the external-mesh reduced model with control region tag $\omega=2$ and Tikhonov parameter $\beta=10^{-3}$. The initialization is $\mathbf{u}_0=\nabla F(\mathbf{0})$ and the stopping criterion is $\|\nabla F(\mathbf{u}_k)\|_U \le \texttt{tol}$ or a cap on iterations.

The Lipschitz constant used by fixed-step methods is estimated via power iteration, yielding $L\approx 1.0010\times10^{-3}$. For constant-parameter Nesterov, the generalized eigenvalues of the Hessian in the $U$-metric satisfy $m\approx 1.0000\times10^{-3}$ and $L\approx 1.3072\times10^{-3}$.

\section{Mesh Independence}
We assess the sensitivity of the optimizer and solution with respect to the mesh resolution. Triangular meshes of the unit square with a circular control region $\omega$ are generated via Gmsh for global sizes $h\in\{0.06,\,0.04,\,0.03,\,0.02,\,0.015,\,0.012,\,0.01\}$. On each mesh, we solve the reduced problem with $\beta=10^{-3}$ and $\omega=2$ (the physical tag of the circle) and compute the unique minimizer $u_*^{(h)}$ by solving the normal equations $Q u = B^\top A^{-1} M\, y_d$ using CG.

For each $h$, we record (i) the final objective value reached by each method and (ii) the iteration counts to reach the stopping criterion. Figure~\ref{fig:mesh-indep} shows that the final objective values stabilize as $h$ decreases, indicating mesh independence of the discretized reduced problem and its numerical solution.
